import SwiftUI

struct ButtonItem: Hashable {
    var color: String
    var title: String
    var description: String
    var icon: String
    init(color: String, title: String, description: String, icon: String) {
        self.color = color
        self.title = title
        self.description = description
        self.icon = icon
    }
}

struct HomeScreen: View {
    var itemList: [ButtonItem] = [ButtonItem(color: "Purple", title: "Scan new", description: "Scanned 483", icon: "plus"), ButtonItem(color: "Purple", title: "Conuterfits", description: "Scanned 483", icon: "plus"), ButtonItem(color: "Purple", title: "Success", description: "Scanned 483", icon: "plus"), ButtonItem(color: "Purple", title: "Directory", description: "Scanned 483", icon: "plus")]

    var body: some View {
        VStack(alignment: .leading) {
            Header()

            ScrollView {
                SectionTitle(title: "Your Insight")

                    ForEach(itemList, id: \.self) { item in
                        SectionItem(items: item).aspectRatio(79/88, contentMode: .fit)
                    }
                }
                SectionTitle(title: "Explore More")
                /* Explore More : Horizontal Slider */
                ScrollView(.horizontal) {
                    HStack {
                        ForEach(moreList, id: \.self) { item in
                            MoreItem(title: item)
                        }
                        .frame(maxWidth: .infinity, alignment: .leading)
                        .listRowInsets(EdgeInsets())
                    }
                }
            }
        }

    struct Header: View {

struct MoreItem: View {

struct SectionItem: View {

    var body: some View {

                    .padding(.top, 8)
            }
        }.onTapGesture {
            print("Route To Self Section")
        }
    }
}

/* Section Title */
